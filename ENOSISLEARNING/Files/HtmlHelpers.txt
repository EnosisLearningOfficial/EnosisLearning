2.Strongly Typed Html Helpers:It Enables these helper methods to directly associate with Model properties in a Strongly Typed View.It Provides helper Methods that you can use only Strongly Typed Views.

namespace MVCTutorial.Models
{
    public class SignUp
    {
        public int UserId { get; set; }
        public string UserName { get; set; }
        public int Password { get; set; }
        public string Gender { get; set; }
        public string Email { get; set; }
        public string Comment { get; set; }
    }
}

@model MVCTutorial.Models.SignUp
@{
    ViewBag.Title = "Index";
}

<h2>Strongly Type Html Helpers</h2>
@using (Html.BeginForm())
{
    @Html.HiddenFor(model => model.UserId)
    <p>@Html.LabelFor(model => model.UserName) &nbsp; @Html.TextBoxFor(model => model.UserName, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Password) &nbsp; @Html.TextBoxFor(model => model.Password, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Gender) &nbsp; @Html.DropDownListFor(model => model.Gender, new SelectList(new[] { "Male", "Female" }), new { @class = "form-select" })</p>
    <p>@Html.LabelFor(model => model.Email) &nbsp; @Html.TextBoxFor(model => model.Email, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Comment) &nbsp; @Html.TextBoxFor(model => model.Comment, new { @class = "form-control" })</p>
}

namespace MVCTutorial.Controllers
{
    public class HtmlHelperController : Controller
    {
        // GET: HtmlHelper
        public ActionResult Index()
        {
            return View();
        }
        [HttpPost]
        public ActionResult Index(Calculation c1)
        {
            int num1 = c1.Num1;
            int num2 = c1.Num2;
            int sum = num1 + num2;
            ViewBag.Result = sum;
            return View();
        }
        public ActionResult About()
        {
            return View();
        }
    }
}
