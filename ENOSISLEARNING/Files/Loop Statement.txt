//For loop 07/08/2025

#include <iostream>
using namespace std;

int main() 
{
	//for loop
    for (int i = 1; i < 5; i++) 
	{
        cout << "For Loop Count: " << i << endl;
    }
    return 0;
}

//while loop
#include <iostream>
using namespace std;

int main() 
{
    int i = 1;
    while (i <= 5) //(1<=5)
	{
        cout << "While Loop Count: " << i << endl;
        i++;//1+1=2---2+1=3--3+1=4--4+1=5--5+1=6
    }
    return 0;
}


//Do-while loop

#include <iostream>
using namespace std;

int main() 
{
    int i = 6;
    do 
	{
        cout << "Do-While Loop Count: " << i << endl;//1
        i++;//1+1=2
    } while (i <= 5);//(2<=5)
    return 0;
}


//continue statement
#include <iostream>
using namespace std;

int main() 
{
    for (int i = 1; i <= 5; i++) 
	{
        if (i == 3) 
		{
            continue;  // Skip the rest when i is 3
        }
        cout << "Continue Example: " << i << endl;
    }
    return 0;
}

//Break statement

#include <iostream>
using namespace std;

int main() 
{
    for (int i = 1; i <= 10; i++) 
	{
        if (i >= 8) 
		{
            break;  // Exit loop when i is 6
        }
        cout << "Break Example: " << i << endl;
    }
    return 0;
}


//goto statement
#include <iostream>
using namespace std;

int main() 
{
    int i = 1;

start:
    if (i <= 5) 
	{
        cout << "Goto Example Count: " << i << endl;
        i++;
        goto start;  // Jump to the label 'start'
    }

    return 0;
}
