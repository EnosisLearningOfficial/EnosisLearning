#include<iostream>
using namespace std;
class Student //Define Class
{
  private:
  int ID;//Data Members
  string Name;
  string Course;
  string Age;

  //Member Functions
  public:
  void SetData()
  {
    cout<<"Enter Id :-"<<endl;
    cin>>ID;
    cout<<"Enter Name :-"<<endl;
    cin>>Name;
    cout<<"Enter Course :-"<<endl;
    cin>>Course;
    cout<<"Enter Age :-"<<endl;
    cin>>Age;
  }
  void ShowData()
  {
    cout<<"Your Id Is:- "<<ID<<endl;
    cout<<"Your Name Is:- "<<Name<<endl;
    cout<<"Your Course Is:- "<<Course<<endl;
    cout<<"Your Age Is:- "<<Age<<endl;
  }
};
int main()
{
  Student s1,s2;
  s1.SetData();
  s1.ShowData();
  s2.SetData();
  s2.ShowData();
   return 0;
}

  }

  //Default Constructor,Parameterless Constructor
  public:

    cout<<"Enter Id :-"<<endl;
    cin>>ID;
    cout<<"Enter Name :-"<<endl;
    cin>>Name;

{
  Student s1,s2;//Constructor Calls automatically

   return 0;

//2.Parameterized Constructor
#include<iostream>
using namespace std;
class Student //Define Class
{
  private:
  int ID;//Data Members
  string Name;
  string Course;
  string Age;

  //Member Functions
  public:
  void SetData()
  {
    cout<<"Enter Id :-"<<endl;
    cin>>ID;
    cout<<"Enter Name :-"<<endl;
    cin>>Name;
    cout<<"Enter Course :-"<<endl;
    cin>>Course;
    cout<<"Enter Age :-"<<endl;
    cin>>Age;
  }
  void ShowData()
  {
    cout<<"Your Id Is:- "<<ID<<endl;
    cout<<"Your Name Is:- "<<Name<<endl;
    cout<<"Your Course Is:- "<<Course<<endl;
    cout<<"Your Age Is:- "<<Age<<endl;
  }

  //Parameterized Constructor
  public:
  Student(int i,string n,string c,string a)
  {
    ID = i;
    Name = n;
    Course = c;
    Age = a;
  }
};
int main()
{
  Student s1 = Student(1,"XYZ","C++","13");
  s1.ShowData();
   return 0;
}
