Templates : 
we can create a function template by using the template keyword.
template<typename T> return_type function_Name(T parameter)
{
	//Code
}

T is a placeholder for a data type.
//1.Generic Method
#include <iostream>
using namespace std;
template<typename T>
T add(T a, T b)
{
    return a+b;
}
int main() {

cout<<add<int>(5,3)<<"\n";
cout<<add<double>(5.89,3.67)<<"\n";
    return 0;
}

//2.C++ Class Templates : You can also use templates to make classes that work with any data type.

template<typename T>
class className
{
	//methods and members using T
}
#include <iostream>
using namespace std;
template<typename T>
class Box
{
  public:
  T value;
  Box(T v)
  {
      value = v;
      cout<<"Value:"<<value<<endl;
  }
};
int main() {

Box<int> intBox(50);
Box<string> strBox("Hello");
    return 0;
}
