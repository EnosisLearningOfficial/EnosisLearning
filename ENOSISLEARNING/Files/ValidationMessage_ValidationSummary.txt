Validation Message and Validation Summary
1.Validation Message :- ValidationMessage() ia an Extension method that is a loosly typed method.it displays a validation Message if an error exist for the specified field in the ModelStateDictionary Object.
<h2>Validation Message</h2>
@using (Html.BeginForm())
{
    <p>
        Enter Name: @Html.TextBox("Name","",new {@class="form-control"})
        @Html.ValidationMessage("Name", new {@class="text-danger"})
    </p>
    <p>
        Enter Age: @Html.TextBox("Age", "", new { @class = "form-control" })
        @Html.ValidationMessage("Age", new { @class = "text-danger" })
    </p>
    <p>
        Enter Email: @Html.TextBox("Email", "", new { @class = "form-control" })
        @Html.ValidationMessage("Email", new { @class = "text-danger" })
    </p>
    <p>
        <input type="submit" name="name" value="Submit" class="btn btn-success" />
    </p>
    @*@ViewBag.Message*@
   @Html.Raw(@ViewData["SuccessMessage"])
}


 public ActionResult Contact()
 {
     return View();
 }
 [HttpPost]
 public ActionResult Contact(string Name,string Age,string Email)
 {
     if(Name.Equals("") == true)
     {
         ModelState.AddModelError("Name", "Name is required");
     }
     if(Age.Equals("") == true)
     {
         ModelState.AddModelError("Age", "Age is required");
     }
     if(Email.Equals("") == true)
     {
         ModelState.AddModelError("Email", "Email is required");
     }
     if(ModelState.IsValid == true)
     {
         //ViewBag.Message = "Form Submitted Successfully";
         ViewData["SuccessMessage"] = "<script>alert('Form Submitted Successfully..!')</script>";
         ModelState.Clear();
     }
     return View();
 }

2.Validation Summary: The ValidationSummary() helper method generates an unordered list(ui Element) of validation message that are in the ModelStateDictionary.
The ValidationSummary can be used to display all the error message for all the fields.it can also be used to display custom error message.

<h2>Validation Summary</h2>
@using (Html.BeginForm())
{
    <p>
        Enter Name: @Html.TextBox("Name","",new {@class="form-control"})
       <b style="color:red;">@ViewData["Name"]</b>
    </p>
    <p>
        Enter Age: @Html.TextBox("Age", "", new { @class = "form-control" })
        <b style="color:red;">@ViewData["Age"]</b>

    </p>
    <p>
        Enter Email: @Html.TextBox("Email", "", new { @class = "form-control" })
        <b style="color:red;">@ViewData["Email"]</b>

    </p>
    <p>
        <input type="submit" name="name" value="Submit" class="btn btn-success" />
    </p>
    <p style="color:red;">@Html.ValidationSummary("", new {@class="text-danger"})</p>
    @*@ViewBag.Message*@
   @Html.Raw(@ViewData["SuccessMessage"])
}

 public ActionResult Contact(string Name,string Age,string Email)

     if(Name.Equals("") == true)

     if(Age.Equals("") == true)
     {
         ModelState.AddModelError("Age", "Age is required");
         ViewData["Age"] = "*";
     }
     if(Email.Equals("") == true)
     {
         ModelState.AddModelError("Email", "Email is required");
         ViewData["Email"] = "*";
     }
     if(ModelState.IsValid == true)
     {
         //ViewBag.Message = "Form Submitted Successfully";
         ViewData["SuccessMessage"] = "<script>alert('Form Submitted Successfully..!')</script>";
         ModelState.Clear();
     }
     return View();
 }
