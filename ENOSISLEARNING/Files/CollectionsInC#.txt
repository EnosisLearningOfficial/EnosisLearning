3.Stack<T> : it is a generic class that arranges the elements of a specified data type using Last In First Out(LIFO) principles.
	1.Push();
	2.Pop();
	3.Peek()
 public static void Main(string[] args)
 {
     //Create a Stack<T> 
     Stack<int> Numbers = new Stack<int>();//Declaration

     //Add Stack<T>
     Numbers.Push(1);
     Numbers.Push(2);
     Numbers.Push(3);
     Numbers.Push(4);
     Numbers.Push(5);
     Numbers.Push(6);

     //Remove
     Numbers.Pop();

     //Peek
     Console.WriteLine("Peeked Value:-" + Numbers.Peek());
     foreach (var i in Numbers)
     {
         Console.WriteLine(i);
     }
 }
4.Queue :it is a generic class that arranges the elements of a specified data type using First In First Out(FIFO) principles.
1.Enqueue
2.Dequeue
        public static void Main(string[] args)
        {
            //Create a Queue<T> 
            Queue<int> Numbers = new Queue<int>();//Declaration

            //Add Queue<T>
            Numbers.Enqueue(1);
            Numbers.Enqueue(2);
            Numbers.Enqueue(3);
            Numbers.Enqueue(4);
            Numbers.Enqueue(5);
            Numbers.Enqueue(6);

            //Remove
            Numbers.Dequeue();

            //Peek
            Console.WriteLine("Peeked Value:-" + Numbers.Peek());
            foreach (var i in Numbers)
            {
                Console.WriteLine(i);
            }
        }
5.SortedList : it is a non-generic collection that contains key/value pair where keys are sorted in an order.
1.Add
2.Access
3.Change
4.Remove(key),RemoveAt(index)
public static void Main(string[] args)
{
    //Create a SortedList 
    SortedList Numbers = new SortedList();//Declaration

    //Add SortedList
    Numbers.Add(1,101);
    Numbers.Add(2,"Abhijeet");
    Numbers.Add(3,32);
    Numbers.Add(4,"Pune");

    //Remove
    Numbers.Remove(1);
    //RemoveAt
    Numbers.RemoveAt(1);
    //Peek
    foreach (var i in Numbers)
    {
        Console.WriteLine(i);
    }
}
