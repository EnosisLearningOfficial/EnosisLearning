using System;
using System.Collections.Generic;
using System.Data.SqlClient;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace AspTutorial
{
    public partial class AspControls : System.Web.UI.Page
    {
        string connectionString = "Data Source =DESKTOP-MPE7G61; Initial Catalog=AspTut;Integrated Security=true;";
        protected void Page_Load(object sender, EventArgs e)
        {

        }

        protected void BtnSubmit_Click(object sender, EventArgs e)
        {
            string name = Nametxt.Text;
            string email = Emailtxt.Text;
            string course = courseDrp.SelectedValue;
            //Response.Write("Name is :- " + name +"\n");
            //Response.Write("Email is :- " + email + "\n");
            //Response.Write("Course is :- " + course + "\n");
            Insert(name, email, course);
        }

        void Insert(string n , string e , string c)
        {
            try
            {
                using (SqlConnection Conn = new SqlConnection(connectionString))
                {
                    string query = "Insert Into Students(Name,Email,Course)values(@Name,@Email,@Course)";
                    using (SqlCommand cmd = new SqlCommand(query, Conn))
                    {
                        cmd.Parameters.AddWithValue("@Name", n);
                        cmd.Parameters.AddWithValue("@Email", e);
                        cmd.Parameters.AddWithValue("@Course", c);
                        Conn.Open();
                        int rows = cmd.ExecuteNonQuery();
                        if (rows > 0)
                        {
                            lblmsg.Text = "Form Submitted Successfully...!";
                            lblmsg.ForeColor = System.Drawing.Color.Green;
                        }
                        else
                        {
                            lblmsg.Text = "Form Submission Failed...!";
                            lblmsg.ForeColor = System.Drawing.Color.Red;
                        }
                        Conn.Close();
                    }
                }
                Nametxt.Text = "";
                Emailtxt.Text = "";
                courseDrp.SelectedIndex = 0;
            }
            catch (Exception ex)
            {
                lblmsg.Text = ex.Message;
            }
        }
    }
}
