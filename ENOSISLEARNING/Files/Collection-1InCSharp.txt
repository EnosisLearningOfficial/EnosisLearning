//C# Collections : In C# Collections are classes that provide an easy way to work with a group of objects.classes provide the implementation of strongly typed entities like lists, stacks etc.

System.Collections.Generic classes  : The System.Collections.Generic classes help us to create a generic collection.This collection does not allow us to store different types of elements.


1.1. List Class : The List<T> class is used to store multiple elements of the same data type that can be accessed using the indexes. We can add, insert and remove elements inside the list. Moreover, we can dynamically change the size of the list.

using System;
using System.Collections.Generic;
namespace CSharpTut
{
    class  Student
    {
        public static void Main(string[] args)
        {
            //Create a list
            List<string> subjects = new List<string>();//Declare List
            List<string> Courses = new List<string>()//Initialize List
            {
                "HTML","CSS","JAVASCRIPT","C#"
            };
            //Access the List Elements :We can access List using index notation []
            Console.WriteLine("Course 1 is : " + Courses[0]);
            Console.WriteLine("Course 2 is : " + Courses[1]);
            Console.WriteLine("Course 3 is : " + Courses[2]);
            Console.WriteLine("Course 4 is : " + Courses[3]);
            //Iterate the List
            for(int i=0;i<Courses.Count;i++)
            {
                Console.WriteLine(Courses[i]);
            }

            foreach(string c in Courses)
            {
                Console.WriteLine(c);
            }
        }
    }
}

3.Remove Elements from the List :We can delete one or more items from List<T> using 2 methods

*Remove() - removes the first occurrence of an element from the given list.
*RemoveAt() - removes the elements at the specified position in the list.
 class  Student
 {
     public static void Main(string[] args)
     {
         //Create a list
         List<string> subjects = new List<string>() { "HTML","CSS","JAVASCRIPT","Bootstrap","CSS","ASP","MVC" };


         // remove the first occurence of "HTML" from the list
         subjects.Remove("CSS");
         // remove the element present at the 2nd index position
         subjects.RemoveAt(2);
         foreach (string subject in subjects)
         {
             Console.WriteLine(subject);
         }

     }
 }

//Stack : A Stack<T> is a generic class that arranges the elements of a specified data type using Last In First Out(LIFO) principles. 

Create a Stack

C# Stack Methods
C# provides 3 major Stack<T> methods. These methods are:

Push() - adds element to the top of the stack
Pop() - removes and returns an element from the top of the stack
Peek() - returns an element from the top of the stack without removing

 class Student
 {
     public static void Main(string[] args)
     {
         //Create a integer type Stack
         Stack<int> stack1 = new Stack<int>();
         //Create a string type Stack
         Stack<string> stack2 = new Stack<string>();

         //Add 5 elements to the stack
         stack1.Push(10);
         stack1.Push(20);
         stack1.Push(30);
         stack1.Push(40);
         stack1.Push(50);

         //Access Stack Elements
         foreach (var i in stack1)
         {
             Console.WriteLine(i);
         }

         //Remove elements from the stack
          stack1.Pop();


         //Access Stack Elements
         foreach (var i in stack1)
         {
             Console.WriteLine(i);
         }


         //Peek method returns the object at the top of the stack without removing it
         Console.WriteLine("Element of Top of Stack" +stack1.Peek());
     }
 }

4.C# Queue<T> : A Queue<T> is a generic class that arranges elements of a specified data type using First In First Out (FIFO) principles. That is, elements that are added first will be removed first.

//Create a Queue in C# 

queue<data_Type> queue_Name = new queue<data_Type>();

C# Queue Methods :
1.Enqueue() - adds an element to the end of the queue
2.Dequeue() - removes and returns an element from the beginning of the queue.
3.Peek() - returns an element from the beginning of the queue without removing
4.Contains() : Check Whether an Element is Present Inside a Queue.We can use the Contains() method to check whether an element is present inside the queue or not.
The method returns True if a specified element exists in the queue. If not it returns False.

class Student
{
    public static void Main(string[] args)
    {
        //Create queue
        Queue<string> List = new Queue<string>();

        //Queue Enqueue() Method
        List.Enqueue("Mumbai");
        List.Enqueue("Pune");
        List.Enqueue("Nagpur");
        List.Enqueue("Kolhapur");
        List.Enqueue("Solapur");
        //Queue Dequeue() Method
        string returnvalue = List.Dequeue();
        Console.WriteLine("Removed Element is :-"+returnvalue);

        //Queue Peek() Method
        string city = List.Peek();
        Console.WriteLine("Peeked Value from the List is :-"+city);
        //Contains() : Check Whether an Element is Present Inside a Queue
        Console.WriteLine(List.Contains("Kolkata"));
        Console.WriteLine(List.Contains("Nagpur"));
        foreach (var item in List)
        {
            Console.WriteLine(item);
        }
    }
}

5.SortedList : A SortedList is a non-generic collection that contains key/value pairs where keys are sorted in an order.

        {
            Console.WriteLine(myList[i]);

        {
            Console.WriteLine(i);
        }
    }
}


6.C# Hashtable : A Hashtable is a non-generic collection that stores key/value pairs that are arranged based on the hashcode of each key.
class Student
{
    public static void Main(string[] args)
    {
        //Create a Hashtable
        Hashtable ht = new Hashtable();

        //Add Elements in Hashtable
        ht.Add("Subject","Math");
        ht.Add("Code",139);
        ht.Add("Marks",100);
        ht.Add("PassMarks",40);

        //Access Hashtable Elements
        Console.WriteLine(ht["Subject"]);
        Console.WriteLine(ht["Code"]);

        //Change Hashtable Elements
        ht["Marks"] = 80;

        //Remove Hashtable Elements
        ht.Remove("Marks");

        //Iterate Hashtable
        foreach (var s in ht.Values)
        {
            Console.WriteLine("values = {0}",s);
        }
    }
}
