//Model class

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace MvcTutorialDemo.Models
{
    public class Student
    {
        public int StudentID { get; set; }

        public string Name { get; set; }

        public string Email { get; set; }

        public string Course { get; set; }
    }
}


//Controller Class

using MvcTutorialDemo.Models;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;

namespace MvcTutorialDemo.Controllers
{
    public class StudentController : Controller
    {
        // GET: Student
        public ActionResult Index()
        {
            var students = new List<Student>
            {
                new Student{StudentID = 1,Name = "Abhijeet Roy",Email="email@gmail.conm",Course="Dot Net"},
                new Student{StudentID = 2,Name = "Roy",Email="email@gmail.conm",Course="Dot Net"}
            };
            return View(students);
        }

        public ActionResult Edit()
        {
            var msg = "Hello Edit is Running";
            return View(msg);
        }
    }
}
