Strongly Typed HTML Helper : It Enables these helper methods to directly associate with Model properties in a Strongly Typed View.
    It Provides helper Methods that you can use only Strongly Typed Views.

1.Html.LabelFor() : Strongly Typed  version Of Html.Label() helper Methods That uses a Lambda Expression as it's parameter,which provides compile time Checking.

2.Html.DisplayNamFor : is Used To Display the names of Model Properties.

3.Html.DisplayFor - is used to display the values of the model properties.

4.Htm.TextBoxFor -  is the Strongly typed version of the Html.TextBox().

5.Html.TextAreaFor - is the Strongly typed version of the Html.TextArea() helper method that generates the same markup as that of the Html.TextAreaFor Helper.


using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace WebApplication1.Models
{
    public class SignUp
    {
        public int UserId { get; set; }
        public string UserName { get; set; }
        public int Password { get; set; }
        public string Gender { get; set; }
        public string Email { get; set; }
        public string Comment { get; set; }
    }
}


@model WebApplication1.Models.SignUp
@{
    ViewBag.Title = "Index";
}
<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.0.2/dist/css/bootstrap.min.css" rel="stylesheet" integrity="sha384-EVSTQN3/azprG1Anm3QDgpJLIm9Nao0Yz1ztcQTwFspd3yD65VohhpuuCOmLASjC" crossorigin="anonymous">
<h2>Index</h2>

@using (Html.BeginForm())
{
    <p>@Html.LabelFor(model => model.UserId) &nbsp; @Html.TextBoxFor(model => model.UserId, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.UserName) &nbsp; @Html.TextBoxFor(model => model.UserName, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Password) &nbsp; @Html.PasswordFor(model => model.Password, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Gender) &nbsp; @Html.DropDownListFor(model => model.Gender, new SelectList(new[] { "Male", "Female" }), "Select Gender", new { @class = "form-select" })</p>
    <p>@Html.LabelFor(model => model.Email) &nbsp; @Html.TextBoxFor(model => model.Email, new { @class = "form-control" })</p>
    <p>@Html.LabelFor(model => model.Comment) &nbsp; @Html.TextAreaFor(model => model.Comment, new { @class = "form-control" })</p>

    <input type="submit" value="Submit" class="btn btn-success" />
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace WebApplication1.Models
{
    public class Calculation
    {
        public int Num1 { get; set; }
        public int Num2 { get; set; }
    }
}


using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;
using WebApplication1.Models;

namespace WebApplication1.Controllers
{
    public class CalculationController : Controller
    {
        // GET: Calculation
        public ActionResult Index()
        {
            return View();
        }
        [HttpPost]
        public ActionResult Index(Calculation c1)
        {
            int num1 = c1.Num1;
            int num2 = c1.Num2;
            int Sum = num1 + num2;
            ViewBag.Sum = Sum;
            return View();
        }
    }
}

@model WebApplication1.Models.Calculation
@{
    ViewBag.Title = "Index";
}
<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.0.2/dist/css/bootstrap.min.css" rel="stylesheet" integrity="sha384-EVSTQN3/azprG1Anm3QDgpJLIm9Nao0Yz1ztcQTwFspd3yD65VohhpuuCOmLASjC" crossorigin="anonymous">
<h2>Index</h2>

@using(Html.BeginForm())
{
    <p>@Html.LabelFor(model => model.Num1) &nbsp; @Html.TextBoxFor(model => model.Num1,new {@class="form-control"})</p>

    <p>@Html.LabelFor(model => model.Num2) &nbsp; @Html.TextBoxFor(model => model.Num2,new {@class="form-control"})</p>

    <input type="submit" value="Submit" class="btn btn-success"/>

    <p class="text-success display-6">Addition Of Two Values Is :- @ViewBag.Sum</p>
}
