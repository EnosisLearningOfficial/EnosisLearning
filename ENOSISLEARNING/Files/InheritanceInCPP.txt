Inheritance - Inheritance allows one class to reuse attributes and methods from another class.
We group the "inheritance concept" into two categories:
1.Derives Class -> Child Class--> sub Class - the class that inherits from another class
2.Base Class -> Parent Class-->Super Class - the class being inherited from

To inherit from a class, use the : symbol.

class Car //Base Class
{
/////Properties
steering,

setdata()
ShowData();

}

//Drived Class
class SuperCar : public Car
{
/////Properties
nos,
navigation,
etc

}


#include <iostream>
using namespace std;
class Car
{
    private:
    string CarName;
    int ModelNo;
    string Color;
    
    public:
    void setData()
    {
        cout<<"Enter Name:-";
        cin>>CarName;
        cout<<"Enter Model no:-";
        cin>>ModelNo;
        cout<<"Enter Color:-";
        cin>>Color;
    }
    void ShowData()
    {
        cout<<"Car Name is :-"<<CarName<<endl;
        cout<<"Model no"<<ModelNo<<endl;
         cout<<"Color"<<Color<<endl;
    }
};
class SuperCar : public Car
{
    private:
    string Nos;
    int Price;
    string Nav;
    
    public:
   void setDataForSP()
    {
        cout<<"Enter Nos:-";
        cin>>Nos;
        cout<<"Enter Price no:-";
        cin>>Price;
        cout<<"Enter Nav:-";
        cin>>Nav;
    }
    void ShowDataForSP()
    {
        cout<<"Car Nos is :-"<<Nos<<endl;
        cout<<"Price"<<Price<<endl;
         cout<<"Nav no"<<Nav<<endl;
    }
};
int main() {
  
    Car c1;
    c1.setData();
    c1.ShowData();
    SuperCar s1;
    s1.setData();
    s1.setDataForSP();
    s1.ShowData();
    s1.ShowDataForSP();
    return 0;
}
